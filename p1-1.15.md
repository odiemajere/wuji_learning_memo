# goto 语句

goto 语句又叫无条件转移语句，可以让程序跳转到任意标记的位置

## 格式

```C
goto |lable......|lable:
```

## 范例

```C
#include <STC15F2K60S2.H>

sbit LED1 = P3^1;
sbit LED2 = P3^0;


void Delay1000ms()		//@5.5296MHz   演示1秒的延时函数
{
	unsigned char i, j, k;
	i = 22;
	j = 3;
	k = 227;
	do
	{
		do
		{
			while (--k);
		} while (--j);
	} while (--i);
}


void Delay_3s(void)
{
	unsigned char i;
	for(i=0;i<3;i++)
	{
		Delay1000ms();
	}
}


void main(void)
{
	unsigned char i;

	LED1 = 0;
	LED2 = 0;

	Delay_3s();


lab1:		for(i=0;i<2;i++)
	{
		LED1 = 1;
		LED2 = 1;
		Delay1000ms();
		LED1 = 0;
		LED2 = 0;
		Delay1000ms();
	}
	goto lab1;
	while(1)
	{
		LED1 = 1;
		LED2 = 1;
		Delay1000ms();
		lab2:
		LED1 = 0;
		LED2 = 0;

	}
}
```
